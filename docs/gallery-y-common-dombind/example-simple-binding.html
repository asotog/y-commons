Example: Simple binding

Jump to Table of Contents

This component provides a simple way to bind the dom to model and viceversa without any configuration, only by defining the model properties to be bind inside of the html

Name:
Last Name:
Married:

Example

HTML

<!-- component include  -->
<script src="/build/gallery-y-common-dombind/gallery-y-common-dombind.js"></script>

<!-- example  -->
<div id="container">
    <form>
        <div>
        Name:
        <input type="text" data-db-bind="name"/>
        </div>
        <div>
        Last Name:
        <input type="text" data-db-bind="lastName"/>
        </div>
        <div>
        Married:
        <input type="checkbox" data-db-bind="married"/>
        </div>
        <button data-db-onclick="submitPerson();">submit</button>
    </form>
</div>

Javascript

YUI().use('gallery-y-common-dombind', function(Y) {

    var dombind = new Y.Common.DomBind({
        container: Y.one('#container'),
        controller: {
            submitPerson: function() {
                alert('Person submitted:' + Y.JSON.stringify(dombind.get('model')))
            }
        }
    });
    
});
