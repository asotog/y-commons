Example: Multiple elements binding

Jump to Table of Contents

Multiple elements can be bind to same model property, reflecting the model updates to every element bind

Name: So your name is ?

Example

HTML

<!-- component include  -->
<script src="/build/gallery-y-common-dombind/gallery-y-common-dombind.js"></script>

<!-- example  -->
<div id="container">
    <form>
        <div>
        Name:
        <input type="text" data-db-bind="name"/>
        So your name is <span data-db-bind="name"></span> ?
        </div>
    </form>
</div>

Javascript

YUI().use('gallery-y-common-dombind', function(Y) {

    var dombind = new Y.Common.DomBind({
        container: Y.one('#container'),
        model: {
            name: 'john doe'
        }
    });
    
});
