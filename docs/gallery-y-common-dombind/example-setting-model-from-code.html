Example: Set model from code

Jump to Table of Contents

The model also can be updated from the code directly and changes are going to be reflected in the ui

Random number is:

Example

HTML

<!-- component include  -->
<script src="/build/gallery-y-common-dombind/gallery-y-common-dombind.js"></script>

<!-- example  -->
<div id="container">
    <form>
        <button data-db-onclick="generateRandom();">generate random</button>
        <div>
        Random number is:
        <input type="text" data-db-bind="random"/>
        </div>
    </form>
</div>

Javascript

YUI().use('gallery-y-common-dombind', function(Y) {

    var dombind = new Y.Common.DomBind({
        container: Y.one('#container'),
        controller: {
            generateRandom: function() {
                dombind.setModel('random', Math.floor(Math.random()*1000));
            }
        }
    });
    
});
